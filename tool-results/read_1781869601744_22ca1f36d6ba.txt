    20‚Üí    20‚Üí    if days_until_thursday == 0 and from_date.hour >= 16:

    62‚Üí    62‚Üídef _bs_delta(spot: float, strike: float, tte: float, iv: float, option_type: str) -> float:
    63‚Üí    63‚Üí    """Calculate Black-Scholes delta."""
    64‚Üí    64‚Üí    if tte <= 0 or iv <= 0:
    65‚Üí    65‚Üí        return 0.0
    66‚Üí    66‚Üí    d1 = (math.log(spot / strike) + (0.05 + 0.5 * iv ** 2) * tte) / (iv * math.sqrt(tte))
    67‚Üí    67‚Üí    if option_type == "CE":
    68‚Üí    68‚Üí        return _normal_cdf(d1)
    69‚Üí    69‚Üí    else:
    70‚Üí    70‚Üí        return _normal_cdf(d1) - 1.0

    73‚Üí    73‚Üídef _bs_price(spot: float, strike: float, tte: float, iv: float, option_type: str) -> float:
    74‚Üí    74‚Üí    """Calculate Black-Scholes option price (simplified)."""
    75‚Üí    75‚Üí    if tte <= 0 or iv <= 0:
    76‚Üí    76‚Üí        intrinsic = max(0, spot - strike) if option_type == "CE" else max(0, strike - spot)
    77‚Üí    77‚Üí        return intrinsic
    78‚Üí    78‚Üí    d1 = (math.log(spot / strike) + (0.05 + 0.5 * iv ** 2) * tte) / (iv * math.sqrt(tte))
    79‚Üí    79‚Üí    d2 = d1 - iv * math.sqrt(tte)
    80‚Üí    80‚Üí    if option_type == "CE":
    81‚Üí    81‚Üí        price = spot * _normal_cdf(d1) - strike * math.exp(-0.05 * tte) * _normal_cdf(d2)
    82‚Üí    82‚Üí    else:
    83‚Üí    83‚Üí        price = strike * math.exp(-0.05 * tte) * _normal_cdf(-d2) - spot * _normal_cdf(-d1)
    84‚Üí    84‚Üí    return max(0.05, round(price, 2))

    87‚Üí    87‚Üídef _bs_greeks(spot: float, strike: float, tte: float, iv: float, option_type: str) -> dict:
    88‚Üí    88‚Üí    """Calculate Black-Scholes greeks."""
    89‚Üí    89‚Üí    if tte <= 0 or iv <= 0:
    90‚Üí    90‚Üí        tte = 1.0 / 365.0
    91‚Üí    91‚Üí        iv = 0.15
    92‚Üí    92‚Üí
    93‚Üí    93‚Üí    d1 = (math.log(spot / strike) + (0.05 + 0.5 * iv ** 2) * tte) / (iv * math.sqrt(tte))
    94‚Üí    94‚Üí    d2 = d1 - iv * math.sqrt(tte)
    95‚Üí    95‚Üí
    96‚Üí    96‚Üí    # Delta
    97‚Üí    97‚Üí    if option_type == "CE":
    98‚Üí    98‚Üí        delta = _normal_cdf(d1)
    99‚Üí    99‚Üí    else:
   100‚Üí   100‚Üí        delta = _normal_cdf(d1) - 1.0
   101‚Üí   101‚Üí
   102‚Üí   102‚Üí    # Gamma
   103‚Üí   103‚Üí    gamma = math.exp(-0.5 * d1 ** 2) / (spot * iv * math.sqrt(tte) * math.sqrt(2 * math.pi))
   104‚Üí   104‚Üí
   105‚Üí   105‚Üí    # Theta (per day)
   106‚Üí   106‚Üí    theta_term1 = -(spot * iv * math.exp(-0.5 * d1 ** 2)) / (2 * math.sqrt(2 * math.pi * tte))
   107‚Üí   107‚Üí    if option_type == "CE":
   108‚Üí   108‚Üí        theta = (theta_term1 - 0.05 * strike * math.exp(-0.05 * tte) * _normal_cdf(d2)) / 365.0
   109‚Üí   109‚Üí    else:
   110‚Üí   110‚Üí        theta = (theta_term1 + 0.05 * strike * math.exp(-0.05 * tte) * _normal_cdf(-d2)) / 365.0
   111‚Üí   111‚Üí
   112‚Üí   112‚Üí    # Vega (per 1% change in vol)
   113‚Üí   113‚Üí    vega = spot * math.sqrt(tte) * math.exp(-0.5 * d1 ** 2) / math.sqrt(2 * math.pi) / 100.0
   114‚Üí   114‚Üí
   115‚Üí   115‚Üí    return {
   116‚Üí   116‚Üí        "delta": round(delta, 4),
   117‚Üí   117‚Üí        "gamma": round(gamma, 4),
   118‚Üí   118‚Üí        "theta": round(theta, 2),
   119‚Üí   119‚Üí        "vega": round(vega, 2),
   120‚Üí   120‚Üí    }

   247‚Üí   247‚Üí        tick_num = self._tick_count[underlying]

   250‚Üí   250‚Üí        self._prev_spot[underlying] = self._spot[underlying]
   251‚Üí   251‚Üí
   252‚Üí   252‚Üí        # Volatility with intraday patterns
   253‚Üí   253‚Üí        vol = config["volatility_daily"] / math.sqrt(375 * 60)  # ~375 min * 60 sec
   254‚Üí   254‚Üí
   255‚Üí   255‚Üí        # Add mean reversion to base price
   256‚Üí   256‚Üí        base = config["base_price"]
   257‚Üí   257‚Üí        current = self._spot[underlying]
   258‚Üí   258‚Üí        mean_reversion = (base - current) * 0.0001
   259‚Üí   259‚Üí
   260‚Üí   260‚Üí        # Add momentum (trend continuation)
   261‚Üí   261‚Üí        momentum = (current - self._prev_spot[underlying]) * 0.05
   262‚Üí   262‚Üí
   263‚Üí   263‚Üí        # Random noise
   264‚Üí   264‚Üí        noise = random.gauss(0, vol * current)
   265‚Üí   265‚Üí
   266‚Üí   266‚Üí        # Occasional larger moves (simulates institutional activity)
   267‚Üí   267‚Üí        if random.random() < 0.02:
   268‚Üí   268‚Üí            noise += random.gauss(0, vol * current * 5)
   269‚Üí   269‚Üí
   270‚Üí   270‚Üí        new_spot = current + mean_reversion + momentum + noise
   271‚Üí   271‚Üí        # Round to 2 decimal places
   272‚Üí   272‚Üí        new_spot = round(new_spot, 2)
   273‚Üí   273‚Üí        self._spot[underlying] = new_spot
   274‚Üí   274‚Üí
   275‚Üí   275‚Üí        spot = self._spot[underlying]
   276‚Üí   276‚Üí        open_price = self._open_spot[underlying]
   277‚Üí   277‚Üí        change = round(spot - open_price, 2)
   278‚Üí   278‚Üí        change_pct = round((change / open_price) * 100, 2) if open_price != 0 else 0.0
   279‚Üí   279‚Üí
   280‚Üí   280‚Üí        now_ms = int(time.time() * 1000)
   281‚Üí   281‚Üí
   282‚Üí   282‚Üí        return {
   283‚Üí   283‚Üí            "symbol": underlying,
   284‚Üí   284‚Üí            "ltp": spot,
   285‚Üí   285‚Üí            "change": change,
   286‚Üí   286‚Üí            "change_pct": change_pct,
   287‚Üí   287‚Üí            "timestamp": now_ms,
   288‚Üí   288‚Üí        }

   294‚Üí   294‚Üí        atm = _atm_strike(spot, config["strike_step"])
   295‚Üí   295‚Üí        step = config["strike_step"]
   296‚Üí   296‚Üí
   297‚Üí   297‚Üí        # Parse expiry to get time to expiry
   298‚Üí   298‚Üí        expiry_dt = datetime.strptime(expiry, "%Y-%m-%d")
   299‚Üí   299‚Üí        now = datetime.now()
   300‚Üí   300‚Üí        tte = max((expiry_dt - now).total_seconds() / (365.25 * 24 * 3600), 1.0 / 365.0)
   301‚Üí   301‚Üí
   302‚Üí   302‚Üí        now_ms = int(time.time() * 1000)

   592‚Üí   592‚Üí

   652‚Üí   652‚Üí

   825‚Üí   825‚Üí    def _simulate_oi(
   826‚Üí   826‚Üí        self,
   827‚Üí   827‚Üí        underlying: str,
   828‚Üí   828‚Üí        instrument_key: str,
   829‚Üí   829‚Üí        strike: float,
   830‚Üí   830‚Üí        atm: float,
   831‚Üí   831‚Üí        option_type: str,
   832‚Üí   832‚Üí        step: int,
   833‚Üí   833‚Üí    ) -> tuple[int, int]:
   834‚Üí   834‚Üí        """
   835‚Üí   835‚Üí        Simulate OI and Change in OI for an option strike.
   836‚Üí   836‚Üí        Returns (oi, change_oi).
   837‚Üí   837‚Üí        """
   838‚Üí   838‚Üí        if instrument_key not in self._oi_state:
   839‚Üí   839‚Üí            # Initialize OI based on distance from ATM
   840‚Üí   840‚Üí            dist_from_atm = abs(strike - atm) / step
   841‚Üí   841‚Üí
   842‚Üí   842‚Üí            # OI is highest near ATM for PE (support) and varies for CE
   843‚Üí   843‚Üí            if option_type == "PE":
   844‚Üí   844‚Üí                # PE OI is highest at ATM and below (support levels)
   845‚Üí   845‚Üí                base_oi = max(10000, int(300000 * math.exp(-dist_from_atm * 0.3)))
   846‚Üí   846‚Üí            else:
   847‚Üí   847‚Üí                # CE OI is highest at ATM and above (resistance levels)
   848‚Üí   848‚Üí                base_oi = max(10000, int(250000 * math.exp(-dist_from_atm * 0.3)))
   849‚Üí   849‚Üí
   850‚Üí   850‚Üí            # Add randomness
   851‚Üí   851‚Üí            open_oi = max(5000, base_oi + random.randint(-base_oi // 4, base_oi // 4))
   852‚Üí   852‚Üí            current_oi = open_oi
   853‚Üí   853‚Üí
   854‚Üí   854‚Üí            self._oi_state[instrument_key] = {
   855‚Üí   855‚Üí                "open_oi": open_oi,
   856‚Üí   856‚Üí                "current_oi": current_oi,
   857‚Üí   857‚Üí            }
   858‚Üí   858‚Üí
   859‚Üí   859‚Üí        state = self._oi_state[instrument_key]
   860‚Üí   860‚Üí
   861‚Üí   861‚Üí        # Update OI with small changes
   862‚Üí   862‚Üí        oi_change = random.randint(-2000, 3000)
   863‚Üí   863‚Üí
   864‚Üí   864‚Üí        # Bias: PE OI tends to build up (support), CE OI also builds (resistance)
   865‚Üí   865‚Üí        if option_type == "PE":
   866‚Üí   866‚Üí            oi_change = int(oi_change * 1.2)  # Slight bullish bias
   867‚Üí   867‚Üí
   868‚Üí   868‚Üí        state["current_oi"] = max(1000, state["current_oi"] + oi_change)
   869‚Üí   869‚Üí        change_oi = state["current_oi"] - state["open_oi"]
   870‚Üí   870‚Üí
   871‚Üí   871‚Üí        return state["current_oi"], change_oi
