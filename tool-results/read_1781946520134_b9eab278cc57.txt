   269‚Üí   269‚Üí    def _compute_7strike_matrix_live(self, symbol: str, expiry: str, spot: float, atm: int, step: int, chain: list) -> Optional[dict]:
   270‚Üí   270‚Üí        """Compute 7-Strike COI PCR Matrix from live option chain data."""
   271‚Üí   271‚Üí        # Build 7-strike window: ATM ¬±3
   272‚Üí   272‚Üí        window_strikes = [atm + i * step for i in range(-3, 4)]
   273‚Üí   273‚Üí
   274‚Üí   274‚Üí        # Map chain by strike for quick lookup
   275‚Üí   275‚Üí        chain_by_strike = {r["strike"]: r for r in chain}
   276‚Üí   276‚Üí
   277‚Üí   277‚Üí        rows = []
   278‚Üí   278‚Üí        for s in window_strikes:
   279‚Üí   279‚Üí            r = chain_by_strike.get(s)
   280‚Üí   280‚Üí            if r:
   281‚Üí   281‚Üí                rows.append({
   282‚Üí   282‚Üí                    "strike": s,
   283‚Üí   283‚Üí                    "ce_coi": r["ce"]["change_oi"],
   284‚Üí   284‚Üí                    "pe_coi": r["pe"]["change_oi"],
   285‚Üí   285‚Üí                    "ce_oi": r["ce"]["oi"],
   286‚Üí   286‚Üí                    "pe_oi": r["pe"]["oi"],
   287‚Üí   287‚Üí                })
   288‚Üí   288‚Üí            else:
   289‚Üí   289‚Üí                rows.append({"strike": s, "ce_coi": 0, "pe_coi": 0, "ce_oi": 0, "pe_oi": 0})
   290‚Üí   290‚Üí
   291‚Üí   291‚Üí        ce_coi_sum = sum(r["ce_coi"] for r in rows)
   292‚Üí   292‚Üí        pe_coi_sum = sum(r["pe_coi"] for r in rows)
   293‚Üí   293‚Üí        coi_pcr = round(pe_coi_sum / ce_coi_sum, 3) if ce_coi_sum != 0 else 1.0
   294‚Üí   294‚Üí
   295‚Üí   295‚Üí        # State determination
   296‚Üí   296‚Üí        if coi_pcr > 1.5 or coi_pcr < 0.6:
   297‚Üí   297‚Üí            state = "ACTIVE"
   298‚Üí   298‚Üí        elif coi_pcr > 1.2 or coi_pcr < 0.8:
   299‚Üí   299‚Üí            state = "ZONE_WATCH"
